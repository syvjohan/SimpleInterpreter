SYSMEMALLOCHEAP 64;

:alias str : #0 = "hej"; /*important to write '\0' otherwise 'g' will be inserted where 'e' is instead of after 'j', str points to 'h'. Observe that it will result in buffer overflow in both cases.*/

:do {
	:regA.add = 1; 
	(&str + :regA.get) += "g";
} :while((&str + :regA.get) != #64);
